\section{Unsoundness in Sum}

\subsection{Unsoundness}

\newcommand{\sone}[0]{\code{S1 }}
\newcommand{\stwo}[0]{\code{S2 }}

The sum state model transformer, $\mmdl_1 \oplus \mmdl_2$ with $\mmdl_1$ and $\mmdl_2$ two valid state models, is currently unsound. In particular, any action that allows flipping the sum from one side to the other is unsound, as it doesn't satisfy frame substraction. This is, for instance, the case with the \code{free} action of the Freeable state model, which is just a type of sum.

The sum state model is defined as $(\mmdl_1 \oplus \mmdl_2).\Sigma \defeq \code{| }\bot \code{ | S1 } \mmdl_1.\Sigma \code{ | S2 } \mmdl_2.\Sigma$, and composition is defined as:

\begin{align*}
	\sigma \cdot \bot &= \sigma\\
	\bot \cdot \sigma &= \sigma\\
	(\sone\st) \cdot (\sone\st') &= \sone(\st\cdot\st')\\
	(\stwo\st) \cdot (\stwo\st') &= \stwo(\st\cdot\st')\\
	\text{undefined}&\text{ otherwise}
\end{align*}

We also remind the frame substraction property, defined as:


\begin{align*}
	&\evalsto{\st\cdot\st_f,e}{o}{\st',v}\Longrightarrow\\
	&\quad(\exists o',v',\st''.~\evalsto{\st,e}{o'}{\st'',v'}~\land\\
	&\qquad(o'\neq \Miss \Rightarrow \st'=\st''\cdot\st_f\land o=o'\land v=v'))
\end{align*}

\newcommand{\proofref}[1]{\textbf{\cal(#1)}}
\newcommand{\proofitem}[1]{\item\textbf{\cal(#1)}~}

\begin{proof}

\pfprop{Sum actions that swap sides are not frame preserving}

Assuming 
\begin{hypvlist}
\hypvitem{pcm1} $\mmdl_1$ is a well formed PCM, $\mmdl_1 \defeq (\Sigma_1, 0_1, \cdot)$
\hypvitem{pcm2} $\mmdl_2$ is a well formed PCM, $\mmdl_2 \defeq (\Sigma_2, 0_2, \cdot)$
\hypvitem{swap} There is an action \code{swap} that accepts no parameters and that, for a state \code{S1 $\st$} or \code{S2 $\st$}, converts it to \code{S2 $\st_2$} or \code{S1 $\st_1$} respectively, with $\st_1$ and $\st_2$ target states of $\mmdl_1$ and $\mmdl_2$. It uses a function ${\code{is\_fully\_owned} : \Sigma\rarr \code{bool}}$ that returns \code{true} if and only if $\st$ is part of $\underline\Sigma$. It is defined as: 

\begin{minted}{ocaml}
let swap !$\st$! =
	match !$\st$! with
	| S1 !$\st$! when is_fully_owned !$\st$! -> ok ((), S2 !$\st_2$!)
	| S2 !$\st$! when is_fully_owned !$\st$! -> ok ((), S1 !$\st_1$!)
	| _ -> miss (MissingState, !$\st$!)
\end{minted}
\end{hypvlist}
	
We want to proove frame substraction does not hold with action \code{swap}.
\begin{description}
	\proofitem{H2} From \proofref{P1} and per definition of sum composition, we have $(\sone\st) \cdot (\sone0_1) = \sone\st$
	\proofitem{H3} Per \proofref{P3} and \proofref{H2}, $\evalsto{(\sone\st) \cdot (\sone\bot_1), \code{swap}~[]}{\Ok}{\stwo\st_2, []}$
	\item Per \proofref{H1} and \proofref{H3}:
		\proofitem{H4}$\exists o',v',\st''.~\evalsto{(\sone\st),\code{swap}~[]}{o'}{\st'',v'}$, and
		\proofitem{H5}$o'\neq\Miss\Rightarrow \stwo\st_2=\st''\cdot\sone0_1\land \Ok=o'\land []=v'$
	\proofitem{H6} Per \proofref{P3} and \proofref{H4}, we have $o'=\Ok$, $\st''=\stwo{\st_2}$ and $v'=[]$
	\proofitem{H7} Per \proofref{H5} and \proofref{H6}, we thus have $\stwo\st_2 = (\stwo\st_2)\cdot(\sone0_1)$
	\proofitem{H8} However, per definition of sum composition, $(\stwo\st_2)\cdot(\sone0_1)$ is undefined, thus $(\stwo\st_2)\cdot(\sone0_1) \neq \stwo\st_2$. Our assumption \proofref{H1} is thus incorrect -- frame substraction does not hold for \code{swap}
\end{description}
\end{proof}

\subsection{Sound Sum}

To define a sound version of sum, we thus need to remove the $0$ element of both sides from the allowed states, resulting in $(\mmdl_1 \oplus \mmdl_2).\Sigma \defeq \code{| }\bot \code{ | S1 } (\mmdl_1.\Sigma~ \backslash \{ 0_1\}) \code{ | S2 } (\mmdl_2.\Sigma~ \backslash \{0_2\})$. This ensure the states $\sone0_1$ and $\stwo0_2$ aren't allowed, avoiding the problem in frame substraction. Additionally, both sides of the sum must also provide a $\code{is\_full\_owned} : \Sigma \rarr \code{bool}$ function, that tests if the state is also part of $\underline\Sigma$.

\begin{proof}$ $\newline
	\textbf{Given } \begin{description}
		\proofitem{P1} $\mmdl_1$ is a well formed PCM, $\mmdl_1 \defeq (\Sigma_1, 0_1, \cdot)$
		\proofitem{P2} $\mmdl_2$ is a well formed PCM, $\mmdl_2 \defeq (\Sigma_2, 0_2, \cdot)$
		\proofitem{P3} There is an action \code{swap} that accepts no parameters and that, for a state \code{S1 $\st$} or \code{S2 $\st$}, converts it to \code{S2 $\st_2$} or \code{S1 $\st_1$} respectively, with $\st_1$ and $\st_2$ target states of $\mmdl_1$ and $\mmdl_2$. It uses a function ${\code{is\_fully\_owned} : \Sigma\rarr \code{bool}}$ that returns \code{true} if and only if $\st$ is part of $\underline\Sigma$. It is defined as: 
\begin{minted}{ocaml}
let swap !$\st$! =
	match !$\st$! with
	| S1 !$\st$! when is_fully_owned !$\st$! -> ok ((), S2 !$\st_2$!)
	| S2 !$\st$! when is_fully_owned !$\st$! -> ok ((), S1 !$\st_1$!)
	| _ -> miss (MissingState, !$\st$!)
\end{minted}
\end{description}

\textbf{Case $\sone\st$, $\sone\st'$}

\begin{description}
	\proofitem{H1} Per definition of sum composition, we have $(\sone\st)\cdot(\sone\st') = \sone(\st\cdot\st')$.
	\proofitem{H2A} If $\code{is\_fully\_owned } \st\cdot\st' = \code{true}$, then as per \proofref{P3} and \proofref{H1}, ${\evalsto{(\sone\st)\cdot(\sone\st'), \code{swap }[]}{\Ok}{\stwo\st_2,[]}}$
	\proofitem{H2B} Per the definition of sum, $\sone0_1$ is not allowed, thus per inextensibility of full states if ${\code{is\_fully\_owned }\st\cdot\st' = \code{true}}$ then $\code{is\_fully\_owned }\st=\code{false}$.
	\proofitem{H2C} Per \proofref{H2B} and \proofref{P3}, we have $\evalsto{\sone\st, \code{swap }[]}{\Miss}{\sone\st, []}$. The outcome is \Miss, thus frame substraction is satisfied.
	\proofitem{H3A} If $\code{is\_fully\_owned } \st\cdot\st' = \code{false}$, then as per \proofref{P3} and \proofref{H1}, \\$\evalsto{(\sone\st)\cdot(\sone\st'), \code{swap }[]}{\Miss}{(\sone\st)\cdot(\sone\st'),[]}$
	\proofitem{H3B} Per \proofref{H3A} it must follow that $\code{is\_fully\_owned } \st = \code{false}$
	\proofitem{H3C} Per \proofref{H3B} and \proofref{P3}, we have $\evalsto{\sone\st, \code{swap }[]}{\Miss}{\sone\st, []}$. The outcome is \Miss, thus frame substraction is satisfied.
\end{description}

\textbf{Case $\sone\st$, $\stwo\st'$}
\begin{description}
	\proofitem{H1} Per definition of sum composition, $\sone\st \cdot\stwo\st'$ is undefined, $\evalsto{\sone\st \cdot\stwo\st',\code{swap }[]}{o}{\st',v}$ is thus false and frame substraction is satified.
\end{description}

\textbf{Case $\sone\st$, $\bot$}

\begin{description}
	\proofitem{H1} Per definition of sum composition, $\sone\st \cdot \bot = \sone\st$
	\proofitem{H2} Per \proofref{H1}, $\evalsto{\sone\st\cdot\bot,\code{swap }[]}{o}{\st',v} \iff \evalsto{\sone\st,\code{swap }[]}{o}{\st',v}$, frame substraction is satisfied.
\end{description}

\textbf{Cases $\stwo\st$ and $\sone\st'$, $\stwo\st'$, $\bot$}
\begin{description}
	\item These cases can be proven analogously to the previous ones.
\end{description}

\textbf{Cases $\bot$ and $\st$}

\begin{description}
	\proofitem{H1} Per definition of sum composition, $\bot\cdot\st=\st$, which is always defined.
	\proofitem{H2} Per \proofref{P3}, $\evalsto{\bot, \code{swap }[]}{\Miss}{\bot, v}$
	\proofitem{H3} Per \proofref{H1} and \proofref{H2}, frame substraction is thus satisfied.
\end{description}

\end{proof}
